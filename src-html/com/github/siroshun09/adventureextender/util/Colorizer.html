017package com.github.siroshun09.adventureextender.util;
018
019import org.jetbrains.annotations.NotNull;
020
021import java.util.regex.Pattern;
022
023/**
024 * An utility class that colorize the legacy-message.
025 */
026public final class Colorizer {
027
028    private static final String EMPTY = "";
029
030    private static final char COLOR_MARK = '&';
031    private static final String COLOR_MARK_STRING = String.valueOf(COLOR_MARK);
032    private static final Pattern COLOR_MARK_PATTERN;
033
034    private static final char COLOR_SECTION = '§';
035    private static final String COLOR_SECTION_STRING = String.valueOf(COLOR_SECTION);
036    private static final Pattern COLOR_SECTION_PATTERN;
037
038    private static final char HEX_MARK = '#';
039    private static final char X = 'x';
040    private static final String COLOR_CODES = "0123456789AaBbCcDdEeFfKkLlMmNnOoRr";
041
042    static {
043        COLOR_MARK_PATTERN = Pattern.compile(COLOR_MARK + "((" + HEX_MARK + "[0-9a-fA-F]{6})|([0-9a-fk-orxA-FK-ORX]))");
044        COLOR_SECTION_PATTERN = Pattern.compile("(?i)" + COLOR_SECTION + "[0-9A-FK-ORX]");
045    }
046
047    private Colorizer() {
048        throw new UnsupportedOperationException();
049    }
050
051    /**
052     * Colorize the string.
053     * <p>
054     * example: {@code "&aHello, &#123456World"} {@literal ->} {@code "§aHello, §x§1§2§3§4§5§6World"}
055     *
056     * @param str the string to colorize
057     * @return the colorized string
058     */
059    public static @NotNull String colorize(String str) {
060        if (str == null || str.isEmpty()) {
061            return EMPTY;
062        }
063
064        if (str.contains(COLOR_MARK_STRING)) {
065            var builder = new StringBuilder(str);
066            return colorize(builder).toString();
067        } else {
068            return str;
069        }
070    }
071
072    /**
073     * Colorize the string.
074     * <p>
075     * example: {@code "&aHello, &#123456World"} {@literal ->} {@code "§aHello, §x§1§2§3§4§5§6World"}
076     * <p>
077     * This method does not support hexadecimal colors represented like {@code #FFF}.
078     *
079     * @param builder the string builder to colorize
080     * @return the colorized string builder
081     */
082    public static @NotNull StringBuilder colorize(StringBuilder builder) {
083        if (builder == null) {
084            return new StringBuilder();
085        }
086
087        int i = builder.indexOf(COLOR_MARK_STRING);
088
089        if (i == -1) {
090            return builder;
091        }
092
093        int l = builder.length();
094
095        while (i != -1 && i + 1 < l) {
096            if (builder.charAt(i + 1) == HEX_MARK) {
097                if (i + 7 < l) {
098                    var hex = builder.substring(i + 2, i + 8);
099
100                    try {
101                        Integer.parseInt(hex, 16);
102                    } catch (NumberFormatException e) {
103                        i = builder.indexOf(COLOR_MARK_STRING, i + 1);
104                        continue;
105                    }
106
107                    builder.setCharAt(i, COLOR_SECTION);
108                    builder.setCharAt(i + 1, X);
109
110                    for (int j = i + 7; j != i + 1; j--) {
111                        builder.insert(j, COLOR_SECTION_STRING);
112                    }
113
114                    l = builder.length();
115                }
116
117                i = builder.indexOf(COLOR_MARK_STRING, i + 1);
118                continue;
119            }
120
121            if (-1 < COLOR_CODES.indexOf(builder.charAt(i + 1))) {
122                builder.setCharAt(i, COLOR_SECTION);
123            }
124
125            i = builder.indexOf(COLOR_MARK_STRING, i + 1);
126        }
127
128        return builder;
129    }
130
131    /**
132     * Strips the given string of color.
133     * <p>
134     * Example: {@code "§aHello, §x§1§2§3§4§5§6World"} {@literal ->} {@code "Hello, World"}
135     *
136     * @param str the string to strip
137     * @return stripped string
138     */
139    public static @NotNull String stripColorCode(String str) {
140        return stripColorCode(str, COLOR_SECTION_PATTERN);
141    }
142
143    /**
144     * Strips the given string of color.
145     * <p>
146     * Example: {@code "&aHello, &#123456World"} {@literal ->} {@code "Hello, World"}
147     *
148     * @param str the string to strip
149     * @return stripped string
150     */
151    public static @NotNull String stripMarkedColorCode(String str) {
152        return stripColorCode(str, COLOR_MARK_PATTERN);
153    }
154
155    private static @NotNull String stripColorCode(String str, @NotNull Pattern pattern) {
156        if (str == null || str.isEmpty()) {
157            return EMPTY;
158        }
159
160        return pattern.matcher(str).replaceAll("");
161    }
162}
