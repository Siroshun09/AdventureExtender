017package com.github.siroshun09.adventureextender.loader.translation;
018
019import com.github.siroshun09.adventureextender.loader.message.MessageLoader;
020import com.github.siroshun09.configapi.common.util.FileUtils;
021import net.kyori.adventure.translation.GlobalTranslator;
022import net.kyori.adventure.translation.TranslationRegistry;
023import org.jetbrains.annotations.NotNull;
024import org.jetbrains.annotations.Unmodifiable;
025
026import java.io.IOException;
027import java.nio.file.Files;
028import java.nio.file.Path;
029import java.util.Collections;
030import java.util.HashSet;
031import java.util.Locale;
032import java.util.Set;
033import java.util.stream.Collectors;
034
035/**
036 * An abstract implementation of {@link TranslationLoader}.
037 */
038public abstract class AbstractTranslationLoader implements TranslationLoader {
039
040    private final Path directory;
041    private final Locale defaultLocale;
042    private final String fileExtension;
043    private final Set<Locale> loadedLocales = new HashSet<>();
044
045    private TranslationRegistry registry;
046
047    /**
048     * Constructor.
049     *
050     * @param directory     the directory path where this loader loads from
051     * @param defaultLocale the default locale
052     * @param fileExtension the file extension to filter files
053     */
054    protected AbstractTranslationLoader(@NotNull Path directory,
055                                        @NotNull Locale defaultLocale, @NotNull String fileExtension) {
056        this.directory = directory;
057        this.defaultLocale = defaultLocale;
058        this.fileExtension = fileExtension;
059    }
060
061    @Override
062    public void load() throws IOException {
063        registry = createRegistry();
064
065        if (!loadedLocales.isEmpty()) {
066            loadedLocales.clear();
067        }
068
069        FileUtils.createDirectoriesIfNotExists(directory);
070
071        saveDefaultIfNotExists();
072
073        loadDefault();
074        loadCustom();
075
076        GlobalTranslator.get().addSource(registry);
077    }
078
079    @Override
080    public void reload() throws IOException {
081        if (registry != null) {
082            GlobalTranslator.get().removeSource(registry);
083        }
084
085        load();
086    }
087
088    @Override
089    public void unload() {
090        if (registry != null) {
091            GlobalTranslator.get().removeSource(registry);
092            loadedLocales.clear();
093            registry = null;
094        }
095    }
096
097    @NotNull
098    @Override
099    public Locale getDefaultLocale() {
100        return defaultLocale;
101    }
102
103    @Override
104    public @NotNull @Unmodifiable Set<Locale> getLoadedLocales() {
105        return Collections.unmodifiableSet(loadedLocales);
106    }
107
108    @Override
109    public @NotNull Path getDirectory() {
110        return directory;
111    }
112
113    /**
114     * Saves default message files if not exists.
115     *
116     * @throws IOException if I/O error occurred
117     */
118    protected abstract void saveDefaultIfNotExists() throws IOException;
119
120    /**
121     * Creates {@link TranslationRegistry}.
122     *
123     * @return new {@link TranslationRegistry}
124     */
125    protected abstract @NotNull TranslationRegistry createRegistry();
126
127    /**
128     * Creates {@link MessageLoader} that matches the format of the file to load.
129     *
130     * @param path the filepath
131     * @return new {@link MessageLoader}
132     */
133    protected abstract @NotNull MessageLoader createMessageLoader(@NotNull Path path);
134
135    private void loadDefault() throws IOException {
136        var defaultFileName = defaultLocale + fileExtension;
137        var defaultFile = directory.resolve(defaultFileName);
138
139        loadFile(defaultFile);
140    }
141
142    private void loadCustom() throws IOException {
143        try (var files = Files.list(directory)) {
144            var languageFiles =
145                    files.filter(Files::isRegularFile)
146                            .filter(p -> !p.toString().endsWith(defaultLocale + fileExtension))
147                            .collect(Collectors.toUnmodifiableSet());
148
149            for (var file : languageFiles) {
150                loadFile(file);
151            }
152        }
153    }
154
155    private void loadFile(@NotNull Path path) throws IOException {
156        var loader = createMessageLoader(path);
157
158        var locale = loader.getLocale();
159
160        if (locale != null) {
161            loader.load();
162            loader.register(registry, locale);
163
164            loadedLocales.add(locale);
165        }
166    }
167}
